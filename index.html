A Fading slideshow w/ next & previous controls.

$('#flower3').slideStrip({autoPlay:false});
$('#prev').click(function(){
    $('#flower3').trigger('previousSlide');
});
$('#next').click(function(){
    $('#flower3').trigger('nextSlide');
});
